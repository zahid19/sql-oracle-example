BEGIN
FOR I IN (
SELECT DD.IMP_BBLC_DETAIL_ID, DD.LAST_UPDATE_DATE, BB.CREATION_DATE
  FROM 
  IEDOC_IMP_BBLC_DETAILS DD, 
  XX_COM_IMP_CNF_BILL_T BB
 WHERE 1=1
 AND DD.IMP_BBLC_DETAIL_ID = BB.IMP_BBLC_DETAIL_ID 
 AND DD.LAST_UPDATE_DATE IS NOT NULL 
 AND BB.CREATION_DATE IS NULL
)
LOOP
UPDATE XX_COM_IMP_CNF_BILL_T BB1
SET BB1.CREATION_DATE = I.LAST_UPDATE_DATE
WHERE BB1.IMP_BBLC_DETAIL_ID  = I.IMP_BBLC_DETAIL_ID  
AND BB1.CREATION_DATE IS NULL ;
COMMIT;
END LOOP;
END;
